/** * 導師進度匯報系統 - 後端 API & 自動化腳本 * 適用於：Google Sheets「工作表1」 */ const SHEET_ID = '1Pvf0uVyTXVtyRDH0nZeExi_xhbpieq_DYvnNJBwmYGQ'; const SHEET_NAME = '工作表1'; // ========================================== // 1. 網頁 API 接口 (處理來自 GitHub 的請求) // ========================================== /** * 處理 GET 請求：讀取資料 */ function doGet(e) { const action = e.parameter.action; if (action === 'getData') { const data = getSheetData(); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput("API 運作正常！請從前端網頁發送請求。"); } /** * 處理 POST 請求：寫入課堂記錄 */ function doPost(e) { try { const params = JSON.parse(e.postData.contents); if (params.action === 'update') { updateRecord(params.rowNum, params.roomStatus, params.progress, params.classStatus, params.remarks); return ContentService.createTextOutput(JSON.stringify({success: true})) .setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({success: false, error: error.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // ========================================== // 2. 資料讀寫邏輯 // ========================================== /** * 從試算表提取 A 至 R 行資料 */ function getSheetData() { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); if (!sheet) return []; const data = sheet.getDataRange().getValues(); const rows = data.slice(1); // 排除第一行標題 return rows.map((row, index) => { return { rowNum: index + 2, // 行號 (用於更新時定位) school: row[0] ? row[0].toString() : '', // A行：學校 lessonNum: row[2] ? row[2].toString() : '', // C行：第幾堂 dateStr: row[3] ? row[3].toString() : '', // D行：日期 dayOfWeek: row[4] ? row[4].toString() : '', // E行：星期 timeStr: row[5] ? row[5].toString() : '', // F行：上課時間 tutors: row[6] ? row[6].toString() : '', // G行：導師 courseName: row[8] ? row[8].toString() : '', // I行：Course Name roomStatus: row[14] ? row[14].toString() : '', // O行：課室情況 progress: row[15] ? row[15].toString() : '', // P行：進度 classStatus: row[16] ? row[16].toString() : '', // Q行：上課情況 remarks: row[17] ? row[17].toString() : '' // R行：備注 }; }); } /** * 寫入資料到指定的 O 至 R 行 */ function updateRecord(rowNum, roomStatus, progress, classStatus, remarks) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); // O行是第15欄，寫入連續4格 (O, P, Q, R) sheet.getRange(rowNum, 15, 1, 4).setValues([[roomStatus, progress, classStatus, remarks]]); return true; } // ========================================== // 3. 試算表自動化與自訂選單 (管理員用) // ========================================== /** * 當試算表打開時，自動建立自訂選單 */ function onOpen() { const ui = SpreadsheetApp.getUi(); ui.createMenu('📅 課堂管理') .addItem('手動更新 Finished 狀態 (L欄)', 'updateFinishedStatus') .addItem('重設狀態 (清空L欄)', 'resetStatus') .addToUi(); } /** * 核心自動化功能：根據 D 欄日期，在 L 欄標註 YES 或 NO * (可手動執行，也可設定「計時器觸發」) */ function updateFinishedStatus() { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); if (!sheet) return; const lastRow = sheet.getLastRow(); if (lastRow < 2) return; // 取得 D 欄 (日期) 到 L 欄 (狀態) 的範圍 const range = sheet.getRange(2, 4, lastRow - 1, 9); const values = range.getValues(); const today = new Date(); today.setHours(0, 0, 0, 0); // 歸零時間以便比較 const results = []; for (let i = 0; i < values.length; i++) { let dateValue = values[i][0]; // 這裡是 D 欄 let status = "NO"; let targetDate = null; if (dateValue instanceof Date) { targetDate = dateValue; } else if (typeof dateValue === 'string' && dateValue !== "") { targetDate = parseChineseDate(dateValue); } if (targetDate && targetDate < today) { status = "YES"; } else { status = "NO"; } results.push([status]); } // 將結果寫回 L 欄 (第 12 欄) sheet.getRange(2, 12, results.length, 1).setValues(results); } /** * 清空 L 欄內容 */ function resetStatus() { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); if (!sheet) return; const lastRow = sheet.getLastRow(); if (lastRow >= 2) { sheet.getRange(2, 12, lastRow - 1, 1).clearContent(); SpreadsheetApp.getUi().alert('L 欄狀態已清空。'); } } /** * 輔助函數：解析中文日期字串 (例如 2026年3月21日) */ function parseChineseDate(dateString) { try { let cleaned = dateString.replace('年', '/').replace('月', '/').replace('日', ''); let d = new Date(cleaned); return isNaN(d.getTime()) ? null : d; } catch (e) { return null; } }
